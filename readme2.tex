\begin{center}
    \textbf{\large 1. Краткая историческая справка развития методов моделирования}
\end{center}

\subsection*{1.1. Этапы эмпирического изучения океанических течений}
История изучения динамики океана связана с мореплаванием. Первые карты течений (Гольфстрим, пассатные течения) появились в эпоху Великих географических открытий. Бенджамин Франклин в XVIII веке объяснил природу высоких температур Гольфстрима переносом теплых вод из тропиков. Важнейший теоретический прорыв совершил Уильям Феррел, описав силу Кориолиса, которая учитывает вращение Земли.

\subsection*{1.2. Развитие численных методов и первые компьютерные модели}
С появлением ЭВМ в 1950-х годах началось численное моделирование. Кирк Брайан (1969) создал первую трехмерную бароклинную модель океана, использующую конечно-разностные методы. В СССР академик Г.И. Марчук разработал метод расщепления, позволивший эффективно решать многомерные задачи на ограниченных ресурсах.

\subsection*{1.3. Совершенствование моделей}
В 1990-х годах отказались от приближения «жесткой крышки», что позволило моделировать уровень моря. Появились модульные модели (например, MOM), позволяющие комбинировать различные физические процессы. Были созданы связанные модели «океан--атмосфера» для прогноза Эль-Ниньо.

\subsection*{1.4. Современное состояние и перспективы}
Современные модели имеют разрешение в несколько километров (вихреразрешающие). Активно развиваются методы ассимиляции данных со спутников и создание «цифровых двойников» океана для оперативного прогнозирования.

\subsection*{3.1. Математическая постановка задачи}
Рассмотрим одномерное нестационарное уравнение теплопроводности как базовую модель переноса:
\begin{equation}
    \frac{\partial T}{\partial t} = \alpha \frac{\partial^2 T}{\partial x^2}, \quad 0 < x < L, \quad t > 0,
\end{equation}
где $\alpha$ -- коэффициент температуропроводности.

Граничные условия первого рода (Дирихле):
\begin{equation}
    T(0, t) = T_1, \quad T(L, t) = T_2.
\end{equation}

\subsection*{3.2. Численный метод: явная конечно-разностная схема}
Аппроксимируем производные разностными отношениями на сетке с шагами $\Delta x$ и $\Delta t$:
\begin{equation}
    \frac{T_i^{n+1} - T_i^n}{\Delta t} = \alpha \frac{T_{i+1}^n - 2T_i^n + T_{i-1}^n}{(\Delta x)^2}
\end{equation}
Отсюда получаем расчетную формулу:
\begin{equation}
    T_i^{n+1} = T_i^n + \sigma \left( T_{i+1}^n - 2T_i^n + T_{i-1}^n \right),
\end{equation}
где $\sigma = \frac{\alpha \Delta t}{(\Delta x)^2}$. Для устойчивости схемы необходимо условие $\sigma \le 0.5$.

\subsection*{3.3. Программная реализация в среде GNU Octave}

\begin{verbatim}
clc; clear; close all;

% 1. Параметры задачи
L = 1.0;            % Длина области
T_final = 5.0;      % Время моделирования
alpha = 0.01;       % Коэффициент теплопроводности

% 2. Сетка
Nx = 50;            % Число узлов
dx = L / (Nx - 1);  % Шаг по пространству
x = linspace(0, L, Nx);

% Шаг по времени из условия устойчивости
dt = 0.4 * (dx^2 / alpha); 
Nt = round(T_final / dt);

% 3. Начальные и граничные условия
T = zeros(1, Nx);   % T(x,0) = 0
T(1) = 100;         % Граничное условие слева
T(end) = 0;         % Граничное условие справа

% 4. Визуализация
figure('Color', 'w'); 
h_plot = plot(x, T, 'b-', 'LineWidth', 2.5); 
hold on; grid on; box on;
xlim([0 L]); ylim([-5 105]);
title_obj = title('Время t = 0.000 с');

% 5. Основной цикл расчета
for n = 1:Nt
    % Расчет внутренних узлов (векторная операция)
    T_inner = T(2:end-1) + (alpha*dt/dx^2) * ...
               (T(3:end) - 2*T(2:end-1) + T(1:end-2));
    
    % Сборка нового слоя с учетом границ
    T = [100, T_inner, 0];
    
    % Обновление графика
    set(h_plot, 'YData', T);
    set(title_obj, 'String', sprintf('Время t = %.3f с', n*dt));
    drawnow; pause(0.01);
end
disp('Моделирование завершено.');
\end{verbatim}
